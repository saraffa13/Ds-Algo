#include<bits/stdc++.h>
using namespace std;

void solve(){
    int n,m;
    cin>>n>>m;
    vector<int>arr(n);
    for(auto &x:arr)cin>>x;
    int dp[n+1][m+1];
    dp[0][0]=1;
    for(int i=1;i<m+1;i++){
        dp[0][i]=0;
    }
    for(int i=1;i<n+1;i++){
        for(int j=0;j<m+1;j++){
            if(j==0){
                dp[i][j]=1;
                continue;
            }
            if(j<arr[i-1]){
                dp[i][j]=dp[i-1][j];
            }else{
                int need = j-arr[i-1];
                if(dp[i-1][j]==1||dp[i-1][need]==1){
                    dp[i][j]=1;
                }else{
                    dp[i][j]=0;
                }
            }
        }
    }
    int ans = dp[n][m];
    if(ans){
        cout<<"YES"<<endl;
    }else{
        cout<<"NO"<<endl;
    }
}

int main () {
    int t;
    cin>>t;
    while(t--){
        solve();
    }
}
